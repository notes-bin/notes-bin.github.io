Mysql数据库

version: "3.9"
services: 
	mysql:
		container_name: mysql-serv
		image: mysql:lastest
		restart: always
		ports:
			- 13306:3306
		networks:
			- mysql-serv-subnet
		environment:
			MYSQL_DATABASE: owner
			MYSQL_USER: owner_admin
			MYSQL_PASSWORD: owner@mysql#123
			MYSQL_ROOT_PASSWORD: root@123#pwd!
			TZ: Asia/Shanghai
		volumes:
			- ./data:/var/lib/mysql
			- /etc/localtime:/etc/localtime:ro
			- /etc/timezone:/etc/timezone:ro
		command:
			- --character-set-server=utf8
			- --collation-server=utf8-bin
networks:
	mysql-serv-subnet:
		name: mysql-serv-subnet
		driver: bridge

Mariadb 数据库

version: "3.9"
services:
	mariadb:
		container_name: mariadb-serv
		image: mariadb:lastest
		restart: always
		ports:
			- 23306:3306
		environment:
			MARIADB_ROOT_PASSWORD: root@123#pwd!
			MARIADB_DATABASE: owner_mariadb
			MARIADB_USER: owner_admin
			MARIADB_PASSWORD: owner@mariadb#123
		volumes:
			- ./data:/var/lib/mysql
			- /etc/localtime:/etc/localtime:ro
			- /etc/timezone:/etc/timezone:ro
		command:
			- --transaction-isolation=READ-COMMITTED
			- --binlog-format=ROW
			- --innodb-file-per-table=1
			- --skip-innodb-read-only-compressed
		networks:
			- mariadb-serv-subnet

networks:
	mariadb-serv-subnet:
		name: mariadb-serv-subnet
		driver: bridge

Redis 数据库

version: "3"
services:
  redis:
    container_name: redis-serv
    image: redis:alpine
    restart: always
    port:
      - 16379:6379
    volumes:
      - ./data:/data
    command:
      - --appendonly yes
      - --databases 64
      - --requirepass redis@pwd@123!
    networks:
      - redis-serv-subnet
  networks:
    redis-serv-subnet:
      name: redis-serv-subnet
      driver: bridge
